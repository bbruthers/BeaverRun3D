095        /**
096         * This method is responcible for building the scene graph. Based on presets x,y, and z it places the lanes and trucks in the scene properly spaced and allined.
097         */
098        public void buildLevel() throws Exception {
099                ConfiguredUniverse cu = new ConfiguredUniverse(
100                        new URL("file:j3dbump.conf")
101                );
102                BoundingBox endZone = new BoundingBox(
103                new Point3d(-x/2, 0.0, -(numberOfLanes + 2) * z - 2),
104                new Point3d(x/2, 10.0, -(numberOfLanes + 1) * z - 2)
105        );
106        
107                
108                
109                
110                BranchGroup bg = new BranchGroup();
111                Background b = new Background(getIC2D(background));
112                b.setImageScaleMode(Background.SCALE_FIT_MAX);
113                b.setApplicationBounds(bs);
114                bg.addChild(b);
115                bg.addChild(setupLights());
116                TransformGroup lanes = new TransformGroup();
117                lanes = addLanes(numberOfLanes);
118                bg.addChild(this.createGrassStrip());
119                bg.addChild(this.createFinishLine());
120                bg.addChild(lanes);
121                FinishLineBehavior flb = new FinishLineBehavior(endZone, this); 
122                flb.setSchedulingBounds(bs);
123                bg.addChild(flb);
124                cu.addBranchGraph(bg);
125                beaverGroup.setCapability(BranchGroup.ALLOW_CHILDREN_READ);
126                beaverGroup.setCapability(BranchGroup.ALLOW_CHILDREN_WRITE);
127                beaverGroup.setCapability(BranchGroup.ALLOW_CHILDREN_EXTEND);
128                beaverGroup.addChild(placeBeaver());
129                beaverGroup.addChild(headsDisplay());
130                cu.addBranchGraph(beaverGroup);
131        }

141        public BranchGroup placeBeaver(){
142                BranchGroup bg = new BranchGroup();
143                bg.setCapability(BranchGroup.ALLOW_DETACH);
144                TransformGroup bt = new TransformGroup();
145                bt.setCapability(TransformGroup.ALLOW_TRANSFORM_READ);
146                bt.setCapability(TransformGroup.ALLOW_TRANSFORM_WRITE);
147                Transform3D bt3d = new Transform3D();
148                Beaver b = new Beaver();
149                Matrix3d roty = new Matrix3d();
150                roty.rotY(Math.toRadians(180.0));
151                Vector3d txtz = new Vector3d(0.0, 1.0, 2.0);
152                double scale = 3.0;
153                bt3d.set(roty, txtz, scale);
154                bt.setTransform(bt3d);
155                bt.addChild(b);
156                BeaverKeyBehavior bkb = new BeaverKeyBehavior(bt);
157                bkb.setSchedulingBounds(bs);
158                BeaverCollisionBehavior bcb = new BeaverCollisionBehavior(b, bg, br3d);
159                bcb.setSchedulingBounds(bs);
160                bg.addChild(bcb);
161                bg.addChild(bkb);
162                bg.addChild(bt);
163                return bg;
164        }

237        private Node placeTruck(double offset, long time) {
238       int startUp = (int)(Math.random() * 1000);
239        
240        if (startUp <= 500){
241                startUp = 500;
242        } else if (startUp > 500){
243                startUp = 1000;
244        }
245                
246                Truck truck = new Truck();
247        TransformGroup tg = new TransformGroup();
248        tg.setCapability(TransformGroup.ALLOW_TRANSFORM_READ);
249        tg.setCapability(TransformGroup.ALLOW_TRANSFORM_WRITE);
250        Transform3D t3d = new Transform3D();
251        t3d.setTranslation(new Vector3d(0.0, 2.75, -offset));
252        tg.setTransform(t3d);
253        tg.addChild(truck);
254        Transform3D t3drppi = new Transform3D();
255        Quat4f[] quats = new Quat4f[5];
256        quats[0] = new Quat4f();
257        quats[1] = new Quat4f();
258        quats[2] = new Quat4f();
259        quats[3] = new Quat4f();
260        quats[4] = new Quat4f();
261        quats[0].set(new AxisAngle4d(0.0, 1.0, 0.0, 0.0));
262        quats[1].set(new AxisAngle4d(0.0, 1.0, 0.0, 0.0));
263        quats[2].set(new AxisAngle4d(0.0, 1.0, 0.0, Math.PI));
264        quats[3].set(new AxisAngle4d(0.0, 1.0, 0.0, Math.PI));
265        quats[4].set(new AxisAngle4d(0.0, 1.0, 0.0, 0.0));
266        Alpha alpha = new Alpha(
267                -1,
268                startUp,
269                4000,
270                time,
271                0,
272                0
273        );
274        RotPosPathInterpolator rppi = new RotPosPathInterpolator(
275                alpha,
276                tg,
277                t3drppi,
278                new float[] { 0.0f, 0.4f, 0.5f, 0.9f, 1.0f},
279                quats,
280                new Point3f[] { 
281                    new Point3f(-x/2.0f, 2.8f, (float) (0.0f - offset)),
282                    new Point3f(x/2f, 2.8f, (float) (0.0f - offset)),
283                    new Point3f(x/2f, 2.8f, (float) (0.0f - offset)),
284                    new Point3f(-x/2f, 2.8f, (float) (0.0f - offset)),
285                    new Point3f(-x/2f, 2.8f, (float) (0.0f - offset))
286                }
287                        
288        );
289        rppi.setSchedulingBounds(bs);
290        tg.addChild(rppi);
291        return tg;
292    }

485        private TransformGroup addLanes(int numberOfLanes){
486                TransformGroup lanes = new TransformGroup();
487                int offset = z;
488                for(int i = 1; i <= numberOfLanes; i++){
489                        lanes.addChild(createLane(offset * i));
490                        lanes.addChild(placeTruck((z/2 + (i) * 8 ), getSpeed(this.truckSpeed)));
491                }
492                return lanes;
493        }
494        private long getSpeed(int speedRange){
495        int number = (int) (Math.random() * 10);
496        while (number == 0){
497                number = (int)(Math.random() * 10);
498        }
499        long speed = (long) (speedRange * number);
500        return speed;
501        }
