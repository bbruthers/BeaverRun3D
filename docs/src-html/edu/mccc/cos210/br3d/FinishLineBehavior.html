010import java.util.Enumeration;

012import javax.media.j3d.Behavior;
013import javax.media.j3d.BoundingBox;
014import javax.media.j3d.WakeupCondition;
015import javax.media.j3d.WakeupOnCollisionEntry;
016import com.sun.j3d.utils.universe.ConfiguredUniverse;
017
018public class FinishLineBehavior extends Behavior {
019
020        BoundingBox finishLine;
021        LevelBuilder lb;
022        WakeupCondition wuc;
023        /**
024         * This will create a new FinishLineBehavior.
025         *@param finishLine this is a reference to the bounding box that is created when the last lane is created. 
026         *@param lb this is a reference to the levelBuilder that created this object.
027         */
028        public FinishLineBehavior(BoundingBox finishLine, LevelBuilder lb) {
029                this.finishLine = finishLine;
030                this.lb = lb;
031        }
032        @Override
033        /**
034         * This is simply the initialization of the behavior as per the Java 3D specifications. It will trigger when the finish line bounding box is collided with by any object.
035         */
036        public void initialize() {
037                wuc = new WakeupOnCollisionEntry(finishLine);
038                wakeupOn(wuc);
039        }
040
041        @SuppressWarnings("rawtypes")
042        @Override
043        /**
044         * When triggered this method will call the level builder's create screen with the win texture.
045         */
046        public void processStimulus(Enumeration arg0) {
047                lb.createScreen("win.jpg");
048                wakeupOn(wuc);
049        }
050
051}
